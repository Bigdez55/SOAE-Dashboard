# Trading Performance Report

Pull P&L data and generate a comprehensive performance dashboard.

---

## Step 1 — Authenticate

```bash
TOKEN=$(curl -s -X POST https://api.elsontrade.com/api/v1/auth/login \
  -H "Content-Type: application/json" \
  -d '{"email":"claude.test@example.com","password":"ElsonTest2026!"}' \
  | python3 -c "import sys,json; print(json.load(sys.stdin).get('access_token',''))")
```

## Step 2 — Portfolio Summary

```bash
curl -s https://api.elsontrade.com/api/v1/portfolio \
  -H "Authorization: Bearer $TOKEN" | python3 -c "
import sys, json
d = json.load(sys.stdin)
print('Equity:', d.get('equity'))
print('Cash:', d.get('cash'))
print('Total P&L:', d.get('total_pnl'))
print('Day P&L:', d.get('day_pnl'))
" 2>/dev/null
```

## Step 3 — Daily P&L (last 30 days)

```bash
curl -s "https://api.elsontrade.com/api/v1/analytics/pnl-history?days=30" \
  -H "Authorization: Bearer $TOKEN" | python3 -c "
import sys, json
data = json.load(sys.stdin)
entries = data if isinstance(data, list) else data.get('data', [])
if entries:
    daily = [e.get('pnl', 0) for e in entries]
    wins = [d for d in daily if d > 0]
    losses = [d for d in daily if d < 0]
    print(f'Trading days: {len(daily)}')
    print(f'Win days: {len(wins)} ({len(wins)/len(daily)*100:.0f}%)')
    print(f'Total P&L: \${sum(daily):.2f}')
    print(f'Avg daily P&L: \${sum(daily)/len(daily):.2f}')
    print(f'Best day: \${max(daily):.2f}')
    print(f'Worst day: \${min(daily):.2f}')
    if wins: print(f'Avg win: \${sum(wins)/len(wins):.2f}')
    if losses: print(f'Avg loss: \${sum(losses)/len(losses):.2f}')
" 2>/dev/null
```

## Step 4 — Strategy Performance

```bash
curl -s "https://api.elsontrade.com/api/v1/analytics/strategy-performance?days=30" \
  -H "Authorization: Bearer $TOKEN" | python3 -c "
import sys, json
data = json.load(sys.stdin)
strategies = data if isinstance(data, list) else data.get('strategies', [])
strategies.sort(key=lambda x: x.get('sharpe', 0), reverse=True)
print('Strategy Performance (by Sharpe):')
for s in strategies[:10]:
    name = s.get('name', s.get('strategy', 'unknown'))
    sharpe = s.get('sharpe', 0)
    wr = s.get('win_rate', 0)
    pnl = s.get('total_pnl', 0)
    print(f'  {name}: Sharpe={sharpe:.2f}, WR={wr:.0%}, P&L=\${pnl:.2f}')
" 2>/dev/null
```
