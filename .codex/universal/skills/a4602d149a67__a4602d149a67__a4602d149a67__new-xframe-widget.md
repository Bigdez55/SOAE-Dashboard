# GEN.OS New XFRAME Widget — C Template

Creates a production-ready XFRAME widget following the exact pattern of
`ui/xframe/widgets/button.c` (the canonical reference implementation).

## Step 1 — Name & Location

File: `ui/xframe/widgets/[widget].c`
Header: `ui/xframe/widgets/[widget].h` (if exposing public API)

## Step 2 — Source Template

```c
/*
 * ui/xframe/widgets/[widget].c — XFRAME [Widget Name] Widget
 *
 * Copyright (c) 2026 GEN.OS Project. All rights reserved.
 * SPDX-License-Identifier: LicenseRef-Proprietary
 *
 * Task: S[N]-XFRAME-W-[ID]
 *
 * PURPOSE:
 *   [2–4 sentences: what this widget does, what user interaction it handles,
 *    which design token categories it uses.]
 *
 * DESIGN:
 *   Static pool of XWIDGET_[WIDGET]_POOL_SIZE ([N]) slots. No heap allocation.
 *   Each instance owns one xframe_node_t surface from the compositor.
 *   State transitions driven by PAL input events forwarded from XDISP.
 *
 * THREAD SAFETY:
 *   xwidget_[widget]_create / destroy: UI thread only.
 *   xwidget_[widget]_set_*: UI thread only.
 *   Paint callbacks: compositor thread — reads state atomically.
 *
 * DEPENDENCIES:
 *   pal.h           — primitive types, spinlock
 *   design_tokens.h — colour / spacing constants
 *   fonts.h         — xfont_render_string, xfont_measure_string
 *   vector.h        — xvec_path_* API
 *
 * No libc. No React. No DOM.
 */

#define PAL_FREESTANDING
#include "../../../pal/include/pal.h"
#include "../runtime/design_tokens.h"
#include "../../../display/xcomp/fonts.h"
#include "../../../display/xcomp/vector.h"

/* ===================================================================
 * CONSTANTS
 * =================================================================== */

#define XWIDGET_[WIDGET]_POOL_SIZE    256U
#define XWIDGET_[WIDGET]_LABEL_MAX    128U

/* Internal state flags */
#define X[W]_FLAG_USED      (1U << 0)
#define X[W]_FLAG_DISABLED  (1U << 1)
#define X[W]_FLAG_HOVERED   (1U << 2)
#define X[W]_FLAG_PRESSED   (1U << 3)
#define X[W]_FLAG_FOCUSED   (1U << 4)
#define X[W]_FLAG_DIRTY     (1U << 5)   /* Needs repaint */

/* ===================================================================
 * STRUCT DEFINITION
 * =================================================================== */

typedef struct {
    uint32_t     id;                               /* 1-based; 0 = unused */
    uint32_t     flags;                            /* X[W]_FLAG_* bitmask */

    char         label[XWIDGET_[WIDGET]_LABEL_MAX]; /* UTF-8 display text */

    /* Layout (compositor surface pixels) */
    int32_t      x;
    int32_t      y;
    uint32_t     w;
    uint32_t     h;

    /* Callbacks */
    void       (*on_[event])(void *userdata);
    void        *on_[event]_ud;

    /* XCOMP surface handle */
    pal_handle_t surface;

    /* Font handle from XFRAME registry */
    void        *font;   /* xfont_t* — opaque */

} xwidget_[widget]_t;

/* ===================================================================
 * STATIC POOL
 * =================================================================== */

static xwidget_[widget]_t g_[widget]_pool[XWIDGET_[WIDGET]_POOL_SIZE];
static uint32_t            g_[widget]_pool_used[XWIDGET_[WIDGET]_POOL_SIZE];
static pal_spinlock_t      g_[widget]_pool_lock;
static int                 g_[widget]_pool_initialized = 0;

/* ===================================================================
 * POOL MANAGEMENT
 * =================================================================== */

static void [widget]_pool_init(void)
{
    if (g_[widget]_pool_initialized) { return; }
    g_[widget]_pool_lock = (pal_spinlock_t)PAL_SPINLOCK_INIT;
    uint32_t i;
    for (i = 0; i < XWIDGET_[WIDGET]_POOL_SIZE; i++) {
        g_[widget]_pool_used[i] = 0;
        g_[widget]_pool[i].id   = 0;
        g_[widget]_pool[i].flags = 0;
    }
    g_[widget]_pool_initialized = 1;
}

static xwidget_[widget]_t *[widget]_pool_alloc(void)
{
    pal_spin_lock(&g_[widget]_pool_lock);
    uint32_t i;
    for (i = 0; i < XWIDGET_[WIDGET]_POOL_SIZE; i++) {
        if (!g_[widget]_pool_used[i]) {
            g_[widget]_pool_used[i] = 1;
            xwidget_[widget]_t *w = &g_[widget]_pool[i];
            w->id    = i + 1U;
            w->flags = X[W]_FLAG_USED;
            pal_spin_unlock(&g_[widget]_pool_lock);
            return w;
        }
    }
    pal_spin_unlock(&g_[widget]_pool_lock);
    return NULL;  /* Pool exhausted */
}

static void [widget]_pool_free(xwidget_[widget]_t *w)
{
    if (!w || w->id == 0) { return; }
    pal_spin_lock(&g_[widget]_pool_lock);
    g_[widget]_pool_used[w->id - 1U] = 0;
    w->id    = 0;
    w->flags = 0;
    pal_spin_unlock(&g_[widget]_pool_lock);
}

/* ===================================================================
 * §1  PUBLIC API
 * =================================================================== */

xwidget_[widget]_t *xwidget_[widget]_create(int32_t x, int32_t y,
                                              uint32_t w, uint32_t h,
                                              const char *label)
{
    [widget]_pool_init();
    xwidget_[widget]_t *widget = [widget]_pool_alloc();
    if (!widget) { return NULL; }

    widget->x = x;  widget->y = y;
    widget->w = w;  widget->h = h;

    /* Copy label safely (no strncpy — freestanding) */
    uint32_t i = 0;
    if (label) {
        while (i < XWIDGET_[WIDGET]_LABEL_MAX - 1U && label[i]) {
            widget->label[i] = label[i]; i++;
        }
    }
    widget->label[i] = '\0';

    widget->flags |= X[W]_FLAG_DIRTY;
    return widget;
}

void xwidget_[widget]_destroy(xwidget_[widget]_t *w)
{
    if (!w) { return; }
    [widget]_pool_free(w);
}

/* ===================================================================
 * §2  PAINT (called from compositor thread)
 * =================================================================== */

void xwidget_[widget]_paint(xwidget_[widget]_t *w)
{
    if (!w || !(w->flags & X[W]_FLAG_DIRTY)) { return; }

    /* Determine colours from design tokens based on state */
    uint32_t bg = (w->flags & X[W]_FLAG_DISABLED)
                  ? DT_COLOR_SURFACE_DISABLED
                  : (w->flags & X[W]_FLAG_PRESSED)
                    ? DT_COLOR_ACCENT_PRESSED
                    : (w->flags & X[W]_FLAG_HOVERED)
                      ? DT_COLOR_ACCENT_HOVER
                      : DT_COLOR_ACCENT;

    /* Draw background */
    xvec_path_t *path = xvec_path_create();
    if (!path) { return; }
    xvec_path_rounded_rect(path, (float)w->x, (float)w->y,
                           (float)w->w, (float)w->h,
                           (float)DT_RADIUS_M);
    /* xvec_fill(path, bg); */   /* TODO: bind to XCOMP surface */
    xvec_path_destroy(path);

    w->flags &= ~X[W]_FLAG_DIRTY;
}
```

## Constraints

- [ ] `#define PAL_FREESTANDING` first
- [ ] Static pool `[N]U` slots, document RAM cost
- [ ] `pool_init()` → `pool_alloc()` → `pool_free()` triad
- [ ] `PAL_SPINLOCK_INIT` macro for lock init
- [ ] `pal_spin_lock()` / `pal_spin_unlock()` for all pool ops
- [ ] Safe label copy (no `strncpy` — freestanding)
- [ ] `DIRTY` flag pattern for repaint optimization
- [ ] Design tokens from `design_tokens.h` (no hardcoded colours)
- [ ] `xvec_path_` prefix for all vector ops
