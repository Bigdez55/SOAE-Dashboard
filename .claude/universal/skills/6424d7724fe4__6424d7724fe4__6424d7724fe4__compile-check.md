# GEN.OS Compile Check — Full Sprint Gate

Run the full compile-check suite locally on macOS. Mirror of CI xenos.yml + display sprint checks.

## Sprint 1: Kernel (22 files)

```bash
KFLAGS="-target x86_64-unknown-none-elf -ffreestanding -fno-stack-protector \
        -fno-pie -mno-red-zone -mno-mmx -mno-sse -mno-sse2 \
        -Wall -Wextra -Werror -O2 -std=c11 -fsyntax-only \
        -Ipal/include -Ikernel/xenos/include -D__x86_64__"

PASS=0; FAIL=0
for f in \
  pal/src/pal_aether.c \
  kernel/xenos/core/main.c kernel/xenos/mm/pmm.c kernel/xenos/mm/vmm.c \
  kernel/xenos/mm/heap.c kernel/xenos/sched/contra_rotation.c \
  kernel/xenos/cap/xkabi_capabilities.c \
  kernel/xenos/drv/intel_igpu.c kernel/xenos/drv/nvme.c \
  init/gensd/main.c init/gensd/service.c init/gensd/supervisor.c \
  init/gensd/journal.c init/gensd/splash.c \
  aetherboot/src/tpm.c aetherboot/src/capability_init.c; do
  if clang $KFLAGS "$f" 2>/dev/null; then
    PASS=$((PASS+1)); echo "  OK  $f"
  else
    clang $KFLAGS "$f"; FAIL=$((FAIL+1)); echo "FAIL  $f"
  fi
done
echo "Sprint 1: $PASS pass, $FAIL fail (target 22/22)"
```
