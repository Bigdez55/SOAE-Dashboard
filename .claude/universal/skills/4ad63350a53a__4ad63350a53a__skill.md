# Elson TB2 Code Quality Audit

Run the fintech-integrity-auditor protocol across the codebase. Produce a prioritized findings report.

---

## Phase 1 — Static Analysis

```bash
# Python: type errors and serious issues
cd backend && python -m pylint app/ --errors-only 2>&1 | head -50

# Python: security issues
cd backend && python -m bandit -r app/ -ll --quiet 2>&1 | head -30

# Frontend: TypeScript errors
cd frontend && npx tsc --noEmit 2>&1 | head -30

# Frontend: ESLint
cd frontend && npx eslint src/ --ext .ts,.tsx 2>&1 | head -30
```

## Phase 2 — Dead Code Detection

```bash
# Unused Python imports
cd backend && python -m autoflake --check --remove-unused-variables \
  --remove-all-unused-imports -r app/ 2>&1 | head -30

# Unreferenced route files
find backend/app/api -name "*.py" | xargs grep -l "router" | while read f; do
  module=$(echo $f | sed 's|.*/||' | sed 's|\.py||')
  grep -r "$module" backend/app/api/api_v1/api.py || echo "POSSIBLY UNREFERENCED: $f"
done
```

## Phase 3 — Financial Safety Checks

```bash
# Check for PII passing to LLM (critical violation)
grep -rn "user_id.*portfolio_context\|email.*eft\|personal.*llm" backend/app/ | grep -v "#"

# Check for float used for money (should be Numeric/Decimal)
grep -rn "Float.*price\|Float.*amount\|Float.*balance\|Float.*pnl" backend/app/models/

# Check for missing redis guards
grep -rn "redis_client\." backend/app/ | grep -v "if redis_client" | grep -v "def " | head -20

# Check for hardcoded secrets
grep -rn "password\s*=\s*['\"].\|api_key\s*=\s*['\"].\|secret\s*=\s*['\"]." backend/app/ | grep -v "os.getenv\|#\|test\|example"
```

## Phase 4 — Test Coverage Gaps

```bash
# Files with no corresponding test
find backend/app/services -name "*.py" | while read f; do
  base=$(basename $f .py)
  if ! find backend -name "test_${base}*" 2>/dev/null | grep -q .; then
    echo "NO TEST: $f"
  fi
done
```

## Phase 5 — Architecture Violations

```bash
# Sync functions that should be async (calling external APIs)
grep -rn "^def.*alpaca\|^def.*yfinance\|^def.*vllm\|^def.*requests" backend/app/services/ | grep -v "#"

# Missing Pydantic v2 patterns
grep -rn "\.dict()\|\.from_orm(" backend/app/ | grep -v "#"

# Direct SQL (should use ORM)
grep -rn "execute.*SELECT\|raw.*query\|text.*SELECT" backend/app/ | grep -v "#\|alembic\|test"
```
